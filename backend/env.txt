# --- REQUIRED: OpenAI API Key ---
# Get one at: https://platform.openai.com/api-keys
OPENAI_API_KEY=your_openai_api_key_here

# --- REQUIRED: AWS Credentials ---
# Step 1: Go to AWS Console → IAM → Users → Create user
# Step 2: Attach policy "AmazonS3ReadOnlyAccess"
# Step 3: Security Credentials → Create Access Key → copy below
AWS_ACCESS_KEY_ID=your_aws_access_key_id_here
AWS_SECRET_ACCESS_KEY=your_aws_secret_access_key_here

# --- REQUIRED: AWS Region where your S3 bucket lives ---
# Examples: us-east-1, us-west-2, eu-west-1
AWS_REGION=us-east-1

# --- REQUIRED: Your S3 bucket name ---
S3_BUCKET_NAME=your_s3_bucket_name_here

# --- LLM Model used to generate answers ---
LLM_MODEL=gpt-4o-mini

# --- Embedding model used to index documents ---
EMBEDDING_MODEL=text-embedding-3-small

# --- How many document chunks to retrieve per query ---
RETRIEVAL_TOP_K=4

# --- Text chunking settings ---
CHUNK_SIZE=1000
CHUNK_OVERLAP=200

# --- ChromaDB vector store directory ---
CHROMA_PERSIST_DIR=./chroma_db

# --- Server ---
HOST=0.0.0.0
PORT=8000
